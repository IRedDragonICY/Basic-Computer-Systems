.MODEL SMALL
.CODE
ORG 100h

TDATA:
	JMP PROSES
	LUSERNAME 	DB 13,10,'USERNAME : $'
	LPASSWORD	DB 13,10,'PASSWORD : $'
	LDITERIMA	DB 13,10,'DITERIMA $'
	LDITOLAK	DB 13,10,'DITOLAK $'

	; menyiapkan memori (var) untuk menerima inputan keyboard
	VUSERNAME	DB 23,?,23 DUP(?)
	VPASSWORD	DB 23,?,23 DUP(?)

PROSES:
	MOV AH,09H
	LEA DX,LUSERNAME
	INT 21H

	MOV AH,0AH
	LEA DX,VUSERNAME
	INT 21H

	MOV AH,09H
	LEA DX,LPASSWORD
	INT 21H

	MOV AH,0AH
	LEA DX,VPASSWORD
	INT 21H

	LEA SI,VUSERNAME
	LEA DI,VPASSWORD
	CLD
	MOV CX,23
	REP CMPSB
	JNE GAGAL

	MOV AH,09H
	LEA DX,LDITERIMA
	INT 21H
	JMP EXIT

GAGAL:
	MOV AH,09H
	LEA DX,LDITOLAK
	INT 21H
	JMP PROSES

EXIT:
	INT 20H
END TDATA
